create database bibliotecapoo;
use bibliotecapoo;
 -- drop database bibliotecapoo;

SET SQL_SAFE_UPDATES = 0;

-- criação das tables

create table livro (
id serial primary key,
titulo text,
autor varchar(100),
sinopse text,
preco double precision,
editora varchar(100),
paginas int,
volume int,
saga varchar(100),
idioma varchar(30),
datapublicacao varchar(10),
disponibilidade int,
categoria text,
versao varchar(50),
classificacaoindicativa int,
isbn varchar(30),
caminhoimagem VARCHAR(255)
);

create table tb_usuario(
id serial,
login varchar(50),
senha varchar(50)
);
